Node.j/ npm
    # Initialize project
    npm init
    npm init -y

    # Install package
    npm install <package>
    npm i <package>

    # Install as dev dependency
    npm i -D <package>

    # Install globally
    npm i -g <package>

    # Install all dependencies
    npm install

    # Remove package
    npm uninstall <package>

    # Update packages
    npm update

    # Check outdated packages
    npm outdated

    # List installed packages
    npm list
    npm list -g --depth=0

    # Run scripts
    npm start
    npm run dev
    npm test
    npm run build

    # Execute package without installing globally
    npx <package>

    # Clear npm cache
    npm cache clean --force


Express.js (Node.js)
    # Create project

    npm init -y

    # Install Express

    npm i -D nodemon

React (Vite)
    # Create React project
    npm create vite@latest

    # Create React project directly
    npm create vite@latest my-app -- --template react

    # Go to project
    cd my-app

    # Install dependencies
    npm install

    # Run development server
    npm run dev

    # Production build
    npm run build

    # Preview production build
    npm run preview

SpringBoot (Maven)
    # Run application
    mvn spring-boot:run

    # Clean project
    mvn clean

    # Compile
    mvn compile

    # Test
    mvn test

    # Package JAR
    mvn package

    # Install to local repository
    mvn install

    # Skip tests
    mvn clean install -DskipTests

    # Dependency tree
    mvn dependency:tree

    # Check dependency updates
    mvn versions:display-dependency-updates
